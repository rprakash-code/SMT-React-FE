import React, { useState, useEffect } from "react";
import { PublicClientApplication, InteractionRequiredAuthError } from "@azure/msal-browser";
import { msalConfig, loginRequest } from "./authConfig"; // Import MSAL config
import axios from "axios";

// Declare MSAL Client Application instance
let pca: PublicClientApplication;

const App = () => {
  const [userInfo, setUserInfo] = useState<any>(null);
  const [error, setError] = useState<string | null>(null);
  const [isMsalInitialized, setMsalInitialized] = useState(false); // Track MSAL initialization

  // Initialize MSAL client app on component mount
  useEffect(() => {
    async function initializeMSAL() {
      try {
        pca = new PublicClientApplication(msalConfig);

        // Check if any account is already authenticated
        const accounts = pca.getAllAccounts();
        if (accounts.length > 0) {
          // If an account is already signed in, try to get user details silently
          await getUserDetails();
        }

        setMsalInitialized(true); // Mark MSAL as initialized
      } catch (error) {
        setError("MSAL Initialization failed: " + error.message);
      }
    }

    initializeMSAL();
  }, []);

  const handleLogin = async () => {
    if (!isMsalInitialized) {
      setError("MSAL is not initialized yet.");
      return;
    }

    try {
      // Perform login using MSAL popup method
      const loginResponse = await pca.loginPopup(loginRequest);
      const accessToken = loginResponse.accessToken; // Access token returned by Microsoft

      // Now call the backend API to get user info (e.g. /api/auth/me)
      const userResponse = await axios.get("http://localhost:5000/api/auth/me", {
        headers: {
          Authorization: `Bearer ${accessToken}`,
        },
      });

      // Store user data in state
      setUserInfo(userResponse.data);
    } catch (err) {
      if (err instanceof InteractionRequiredAuthError) {
        setError("Authentication requires interaction.");
      } else {
        setError("Login failed! " + err.message);
      }
    }
  };

  const getUserDetails = async () => {
    try {
      const accounts = pca.getAllAccounts();
      if (accounts.length > 0) {
        const currentAccount = accounts[0];
        const accessTokenResponse = await pca.acquireTokenSilent({
          ...loginRequest,
          account: currentAccount,
        });

        // Use access token to fetch user details from backend or Graph API
        const userResponse = await axios.get("http://localhost:5000/api/auth/me", {
          headers: {
            Authorization: `Bearer ${accessTokenResponse.accessToken}`,
          },
        });

        // Store user data
        setUserInfo(userResponse.data);
      }
    } catch (err) {
      setError("Error getting user details: " + err.message);
    }
  };

  return (
    <div>
      <h2>Login with Microsoft</h2>
      {!userInfo ? (
        <button onClick={handleLogin}>Login</button>
      ) : (
        <div>
          <h3>Welcome {userInfo.displayName}</h3>
          <p>Email: {userInfo.mail}</p>
        </div>
      )}

      {error && <div>{error}</div>}
    </div>
  );
};

export default App;
